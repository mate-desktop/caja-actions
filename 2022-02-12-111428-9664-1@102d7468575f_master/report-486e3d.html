677/*
678 * object may embed a list of - possibly negated - mimetypes
679 * each file of the selection must satisfy all conditions of this list
680 * an empty list is considered the same as '*' or '* / *'
681 *
682 * most time, we will just have '*' - so try to optimize the code to
683 * be as fast as possible when we don't filter on mimetype
684 *
685 * mimetypes are of type : "image/ *; !image/jpeg"
686 * file mimetype must be compatible with at least one positive assertion
687 * (they are ORed), while not being of any negative assertions (they are
688 * ANDed)
689 *
690 * here, for each mimetype of the selection, do
691 * . match is FALSE while this mimetype has not matched a positive assertion
692 * . nomatch is FALSE while this mimetype has not matched a negative assertion
693 * we are going to check each mimetype filter
694 * while we have not found a match among positive conditions (i.e. while !match)
695 * we have to check all negative conditions to verify that the current
696 * examined mimetype never match these
697 */

*) g_type_check_instance_cast ((GTypeInstance*) (object), (( na_ifactory_object_get_type
())))))), "na-factory-data-mimetypes" ));
709 GSList *im;
710 GList *it;
